using PT4;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;
using System.Xml.Linq;

namespace PT4Tasks
{
    public class MyTask : PT
    {
        public static void Solve()
        {
            Task("LinqXml49");
            string name = GetString();
            string s = GetString();
            XDocument d = XDocument.Load(name);
            TimeSpan t = d.Descendants()
                .Select(e => (TimeSpan?)e.Attribute(s) ?? new TimeSpan(0,0,0))
                .Aggregate(TimeSpan.Zero, (a, e) => a + e);
            //Show(t);

            d.Root.SetAttributeValue("total-time", t);
            d.Save(name);
        }
    }
}
