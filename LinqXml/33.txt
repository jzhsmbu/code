using PT4;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;
using System.Xml.Linq;

namespace PT4Tasks
{
    public class MyTask : PT
    {
        public static void Solve()
        {
            Task("LinqXml33");
            string name = GetString();
            XDocument d = XDocument.Load(name);
            foreach (var e in d.Root.Elements())
            {
                if (e.Attributes().Count() > 0)
                {
                    XElement attr = new XElement("attr");
                    attr.ReplaceAttributes(e.Attributes());
                    e.Add(attr);
                    e.RemoveAttributes();
                }

            }
            d.Save(name);

        }
    }
}
