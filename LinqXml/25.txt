using PT4;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;
using System.Xml.Linq;

namespace PT4Tasks
{
    public class MyTask : PT
    {
        public static void Solve()
        {
            Task("LinqXml25");
            string name = GetString();
            XDocument d = XDocument.Load(name);
            foreach(var e1 in d.Root.Elements())
            {
                if (e1.Attributes().Count() > 1)
                    e1.RemoveAttributes();
            }
            foreach(var e2 in d.Root.Elements().Elements())
            {
                if (e2.Attributes().Count() > 1)
                    e2.RemoveAttributes();
            }
            d.Save(name);
        }
    }
}
