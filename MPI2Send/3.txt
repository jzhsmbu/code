#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPI2Send3");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    int a;
    int aa[4] = { 0 };
    if (rank > 0)
    {

        for (int k = 0; k < 4; k++)
        {
            pt >> a;
            aa[k] = a;
        }

        MPI_Send(aa, 4, MPI_INT, 0, 0, MPI_COMM_WORLD);
    }

    else
    {
        for (int i = 1; i < size; ++i)
        {
                int b[4]={0};
                MPI_Recv(b, 4, MPI_INT, i, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);

                for (int i = 0; i < 4; i++)
                {
                    pt << b[i];
                }
        }
    }
}
