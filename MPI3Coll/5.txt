#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPI3Coll5");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    int* rcounts = new int[size];
    int sum=0;
    for (int i = 0; i < size; i++)
    {
        rcounts[i] = i + 2;
        sum = sum + rcounts[i];
    }
    //Show(sum);

    int* dis = new int[size];
    for (int i = 0; i < size; i++)
    {
        if (i == 0)
            dis[i] = 0;
        else
            dis[i] = dis[i - 1] + rcounts[i - 1];
    }
    int aa[100]={0}, res[100]={0};
    for (int i = 0; i < rank+2; i++)
    {
        int a;
        pt >> a;
        aa[i] = a;
    }
    MPI_Gatherv(aa, rank+2, MPI_INT, res, rcounts, dis, MPI_INT, 0, MPI_COMM_WORLD);
    if (rank == 0)
    {
        for (int i = 0; i < sum; i++)
        {
            pt << res[i];
        }
    }
}
