for (p in 1:3)
{
  for (q in 1:3)
  {
    tfx <- tfest(Y, X, delay = 5, p = p, q = q, um.x = umx, um.y = umy)
    tfmy <- tfm(Y, inputs = tfx, noise = um(ar = 3))
    smr <- summary(tfmy)
    cat("p = ", p , "q = " , q)
    cat(" aic = ",smr$aic)
    cat(" bic = ",smr$bic)
    cat(" rss = ",smr$rss)
    cat('\n')
  }    
}
