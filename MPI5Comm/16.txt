#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPI5Comm16");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    int K = size;
    int N = size / 2;
    MPI_Comm new_comm;
    int dims[] = { N,2 };
    int periods[] = { 0,0 };

    MPI_Cart_create(MPI_COMM_WORLD, 2, dims, periods, 0, &new_comm);

    MPI_Comm comm_sub;
    int remain_dims[] = {1,0};
    MPI_Cart_sub(new_comm, remain_dims, &comm_sub);
    MPI_Comm_size(comm_sub, &size);
    MPI_Comm_rank(comm_sub, &rank);
    
    double A;
    if (rank == 0)
    {
        pt >> A;
    }

    MPI_Bcast(&A, 1, MPI_DOUBLE, 0, comm_sub);
    pt << A;
}
