#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPI5Comm26");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    MPI_Status s;
    double A,B;
    pt >> A;

    MPI_Comm new_comm;
    int K = size;
    int N = K / 2;
    int dims[] = { 2,N };
    int periods[] = { 0,1 };
    MPI_Cart_create(MPI_COMM_WORLD, 2, dims, periods, 0, &new_comm);

    int source, dest;
    MPI_Cart_shift(new_comm, 1, 1, &source, &dest);
    Show(source);
    Show(dest);

    MPI_Sendrecv(&A, 1, MPI_DOUBLE, dest, 0, &B, 1, MPI_DOUBLE, source, 0, new_comm, &s);
    pt << B;
}
