#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPI6File12");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    char name[12];
    if (rank == 0)
    {
        pt >> name;
    }
    MPI_Bcast(name, 12, MPI_CHAR, 0, MPI_COMM_WORLD);

    int a;
    MPI_Comm new_comm;
    pt >> a;
    MPI_Comm_split(MPI_COMM_WORLD, a > 0 ? 1 : MPI_UNDEFINED, rank, &new_comm);
    if (new_comm == MPI_COMM_NULL)
        return;

    MPI_Comm_rank(new_comm, &rank);
    //Show(rank);
    MPI_Status s;
    MPI_File f;
    MPI_File_open(new_comm, name, MPI_MODE_WRONLY | MPI_MODE_CREATE, MPI_INFO_NULL, &f);

    double b = rank;
    MPI_File_write_at_all(f, (a-1) * sizeof(double), &b, 1, MPI_DOUBLE, &s);

    MPI_File_close(&f);
}
