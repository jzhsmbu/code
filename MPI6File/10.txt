#include "pt4.h"
#include "mpi.h"

void Solve()
{
    Task("MPI6File10");
    int flag;
    MPI_Initialized(&flag);
    if (flag == 0)
        return;
    int rank, size;
    MPI_Comm_size(MPI_COMM_WORLD, &size);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);

    char name[12];
    if (rank == 0)
    {
        pt >> name;
    }
    MPI_Bcast(name, 12, MPI_CHAR, 0, MPI_COMM_WORLD);


    MPI_File f;
    MPI_File_open(MPI_COMM_WORLD, name, MPI_MODE_WRONLY | MPI_MODE_CREATE, MPI_INFO_NULL, &f);

    int count = 0;
    int* a = new int[rank];
    //a[0] = 0;
    for (int i = 0; i < rank; i++)
    {
        pt >> a[i];
    }

    for (int i = 0; i < rank; i++)
    {
        count = count + i;
    }
    Show(count);

    MPI_File_seek(f, count* sizeof(int), MPI_SEEK_SET);

    MPI_File_write_all(f, a, rank, MPI_INT, MPI_STATUS_IGNORE);

    MPI_File_close(&f);
}
